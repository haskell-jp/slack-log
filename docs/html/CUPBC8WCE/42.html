-- 例: concatNumber' [1..3] = [[1,2,3],[1,23],[12,3],[123]]
concatNumber' :: [Int] -> [[Int]]
concatNumber' xs = aux xs id [] where
  aux [] _ _          = []
  aux [x] f cum       = f [x] : cum
  aux (x:y:zs) f cum  = aux (y:zs) (f . (x:)) (aux ((10*x+y):zs) f cum)
