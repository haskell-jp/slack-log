buildMap :: (Ord key) => ([Maybe key], [Maybe val]) -> Maybe (Map key val)
buildMap = \ case
  (Just k : ks, Just v : vs) -> Map.insert k v <$> buildMap (ks, vs)
  (Nothing : _, _)           -> Nothing
  (_, Nothing : _)           -> Nothing
  ([], [])                   -> Just (Map.empty)
  (_:ks, [])                 -> const (Map.empty) <$> buildMap (ks, [])
  (e@[], _:vs)               -> asTypeOf (Map.empty) <$> buildMap ([], vs)
