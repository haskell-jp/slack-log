    ) where

someFunc :: IO ()
someFunc = do
  credentialLoadX509 "./localhost.pem" "./localhost-key.pem" >>=
    \case

        void $ forkIO $ serve (makeServerSettings c Nothing) "localhost" "8080" $ \(ctx, _) -> establish ctx
      _ ->
        putStrLn "Error"
  -- 終了させないためだけの処理
  _ <- getLine
  pure ()

  r <- recv ctx
