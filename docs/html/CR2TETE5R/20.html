type A<T> = { a: T }
type B<F> = { b: F }

type C = A<B<number>>

type GetT<AA extends A<any>> = AA extends A<infer T> ? T : never
type GetF<BB extends B<any>> = BB extends B<infer F> ? F : never

type D = GetF<GetT<C>> // number
