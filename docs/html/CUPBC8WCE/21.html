GHC 8.10.1のGHCiでできました。以下を貼り付けてね！
:set -XTypeOperators

import Data.Functor.Compose

type (.) = Compose
:k (.)

:set -XDerivingVia
:{
newtype ReaderIO env a = ReaderIO (env -> IO a)
    deriving (Functor, Applicative) via ((->) env . IO)
:}
act = (ReaderIO putStrLn *> ReaderIO putStrLn)
ReaderIO act' = act
act' "hello"
