group :: Monad m => (a -> Bool) -> ConduitT a [a] m ()
group f = loop id
  where
    loop rest = do
        mx <- await
        case mx of
          Nothing -> case rest [] of
            [] -> return ()
            l  -> yield l
          Just x
            | f x       -> yield (rest []) >> loop id
            | otherwise -> loop (rest . (x:))
