main = print $ as ! n -- => 結果 5枚（43円硬貨 1枚、14円硬貨 4枚）
as = array (0, n) ((0, 0) : [(i, 1 + minimum [as ! (i - j) | j <- coins, j <= i]) | i <- [1 .. n]])
