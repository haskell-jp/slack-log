pa = P.switch collect combine
  where
    collect = proc x -> do
      lRev <- P.accum [] -< (:) <$> x
      sum' <- P.accum 0 -< (+) <$> x
      ed <- P.onEnd -< x
      P.muted &&& returnA -< (reverse lRev, sum') <$ ed

      returnA -< (, sum') <$> x
