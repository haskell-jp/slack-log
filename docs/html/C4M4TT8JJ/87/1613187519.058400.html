(!) :: (Show? key, Ord key) => Map key value -> key -> value
(!) m key = case lookup key m of
  Just value -> value
  Nothing -> case show? of
    Nothing -> error $ "no value for key"
    Just fshow -> error $ "no value for key: "++ fshow key

>>> empty ! "hoge"

>>> newtype T = T Int deriving(Eq,Ord)
>>> empty ! (T 10)
