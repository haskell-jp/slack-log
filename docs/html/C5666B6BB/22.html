{-# LANGUAGE ConstraintKinds, TypeFamilies #-}
import Data.Constraint
data Foo
data Bar

type family IsFooOrBar x :: Constraint where
  IsFooOrBar Foo = ()
  IsFooOrBar Bar = ()
  IsFooOrBar x = Bottom

someFun :: (IsFooOrBar a, Monad m) => m a
someFun = undefined
