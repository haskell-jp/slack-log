Entryから値を取り出すことができました。
  button <- new Gtk.Button [ #label := "Run" ]

  on button #clicked (do
                          buffer <- Gtk.entryGetBuffer entry
                          buffer2 <- Gtk.entryBufferGetText buffer
                          set msg [ #sensitive := False, #label := buffer2 ]
                     )

  #add box button
